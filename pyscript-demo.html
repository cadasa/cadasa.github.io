Let's plot linear regression of some random numbers using numpy, pandas and seaborn.

import numpy as np import pandas as pd import seaborn as sns sns.set_theme(style="ticks") np.random.seed(123) # Choose the "true" parameters. m_true = -0.9594 b_true = 4.294 f_true = 0.534 # Generate some synthetic data from the model. N = 50 x = np.sort(10 * np.random.rand(N)) yerr = 0.1 + 0.5 * np.random.rand(N) y = m_true * x + b_true y += np.abs(f_true * y) * np.random.randn(N) y += yerr * np.random.randn(N) df = pd.DataFrame({"x": x, "y": y}) sns.lmplot(x='x', y='y', data=df)
